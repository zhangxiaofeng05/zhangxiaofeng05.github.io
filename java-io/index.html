public class IoTest {
    public static void main(String args[]){
        try {
            File file = new File("aa.txt");
            InputStream is = new FileInputStream(file);
            InputStreamReader isr = new InputStreamReader(is,"utf-8");
            BufferedReader br = new BufferedReader(isr);

            is.close();

            e.printStackTrace();
        }catch (Exception e){
            e.printStackTrace();
        }
    }
}
